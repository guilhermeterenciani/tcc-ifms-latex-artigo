\def\@maketitle{%
	\newpage
	\null
	\vskip 2em%
	\begin{center}%
		\let \footnote \thanks
		% {\LARGE \@title \par}%
		{\large \MakeUppercase{\bfseries{\@title}}\par}%
		\vskip 1em%

%	  {\normalsize
%		  % \lineskip .5em%
%		  \begin{tabular}[t]{c}%
%			\@author
%		  \end{tabular}\par
%	  }%
	\end{center}%
	\par
	
	\begin{flushright}
	\autorI - \autorIEmail
	\\
	\autorII - \autorIIEmail
	\end{flushright}
	
	\vskip 1.5em}
\makeatother
